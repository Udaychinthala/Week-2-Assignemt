3.select modulename from courseandmodules inner join modules on courseandmodules.modulecode=modules.modulecode where coursecode=100;
4.select modulename,coursecode from courseandmodules inner join modules on courseandmodules.modulecode=modules.modulecode where coursecode=100;
5.select count(modulename) as NumberofModules from courseandmodules inner join modules on courseandmodules.modulecode=modules.modulecode where coursecode=100;
6.select courseandmodules.coursecode,count(modulename) as NumberofModules from courseandmodules inner join modules on courseandmodules.modulecode=modules.modulecode GROUP BY courseandmodules.coursecode;
7.select title,fees from courses order by fees desc limit 1;
8.select studentid,studentname,students.batchcode,place,phone from students inner join batches on students.batchcode=batches.batchcode where batches.batchcode=3001;

13.select modulename from modules inner join courseandmodules on modules.modulecode=courseandmodules.modulecode inner join courses on courseandmodules.coursecode=courses.coursecode where title='ORACLE DBA' group by modules.modulename;
